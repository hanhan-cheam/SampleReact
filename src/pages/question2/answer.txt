const targetRow = tableData.filter(x => x.id === targetRowId).map(y => {
    return {
        ...y,
        skuCode: inputFields.find(x => x.name === "skuCode")?.value as string,                
        skuDescription:inputFields.find(x => x.name === "skuDescription")?.value as string,
        color:inputFields.find(x => x.name === "color")?.value as string,
        batchNo:inputFields.find(x => x.name === "batchNo")?.value as string 
    }
})
const nonTargetRow = tableData.filter(x => x.id !== targetRowId)

const newRows = targetRow.concat(nonTargetRow).sort((a, b) => a.id - b.id)

setTableData(newRows)
