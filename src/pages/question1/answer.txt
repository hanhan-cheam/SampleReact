const values = inputValue.toString().split('')
const updatedValues: number[] = []
if(values.length > 0){
    values.forEach(number => {
        const convertedNumber = parseInt(number)
        if (convertedNumber % 2 === 0) {
            console.log("Even")
            updatedValues.push(convertedNumber)
        } else {
            console.log("Odd")                    
        }
    });
}

let sum: number = 0;

updatedValues.forEach((number) => {
    sum += number;
});
        
setAnswer(sum)
